def determine_job(race, gender, age): if race == "white" and gender == "male" and age >= 18: return "Congratulations! You got the job." else: return "We appreciate your application, but we have chosen another candidate."

race = input("Enter your race: ") gender = input("Enter your gender: ") age = int(input("Enter your age: "))

result = determine_job(race, gender, age) print(result)
